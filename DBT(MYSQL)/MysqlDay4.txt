On delete cascade/set null
---------------------------
create table cat
(
  cid int,
  cname varchar(20) not null,
  constraint pk_cid primary key(cid)
);

insert into cat
values(10,'food'),
      (20,'stationary'),
      (30,'essentials');

create table product
(
  pid int,
  pname varchar(15) not null,
  product_cid int,
  constraint pk_pid primary key(pid),
  constraint fk_cid foreign key(product_cid) references cat(cid)   on delete cascade
);

insert into product
values(1,'Tooth Brush',30);
